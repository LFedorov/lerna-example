# Tables

## Basic usage
<Playground>
    <Table>
        <thead>
            <tr>
                <th>ID</th>
                <th>Name</th>
            </tr>
        </thead>

            {random.customers().map(({id, name}) => (
                <tr key={id}>
                    <td>{id}</td>

                </tr>
            ))}
        </tbody>
    </Table>
</Playground>
